# Pinout
Below pinout is for ESP32 version.

## LEDs
Badge Complete = GPIO2
Badge Bonus = GPIO4

Score 1s = GPIO5
Score 2s = GPIO6
Score 4s = GPIO7
Score 8s = GPIO15

RPS Complete = GPIO8
RPS Rock = GPIO16
RPS Paper = GPIO17
RPS Scissors = GPIO18

Simon Complete = GPIO46
Simon Up = GPIO3
Simon Right = GPIO9
Simon Left = GPIO11
Simon Down = GPIO10

DTMF Complete = GPIO43
DTMF Bonus = GPIO44

HiLo Complete = GPIO40
HiLo Hi = GPIO42
HiLo Lo = GPIO41

Kode Complete = GPIO39


## Buttons
Up = GPIO38
Right = GPIO47
Left = GPIO45
Down = GPIO48

Select = GPIO14
Start = GPIO21

A = GPIO12
B = GPIO13
